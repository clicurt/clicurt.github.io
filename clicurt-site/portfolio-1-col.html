const treeReducer = (
  state = {
    data: []
  },
  action
) => {
  switch (action.type) {
    case "GET_TREE_DATA":
      return {
        ...state,
        data: createDataTree(action.payload)
      };
    default:
      return state;
  }
  // return state;

const createDataTree = dataset => {
  let hashTable = Object.create(null);
  let children = [];
  let arr = dataset.label;
  children = [
    ...children,
    { name: dataset.IMEO_SE },
    { name: dataset.LABEL_STATUS }
  ];

  arr.map(aData => {
    return hashTable[aData.name] = {
      ...aData,
      children: []
    };
  });

  let dataTree = [];
  let names = [];
  let basename_level = [];
  // let label = Object.create(null);
  arr.map(data => {
    names.push(data.name);
    basename_level.push(data.basename_level);

    let flag = false;
    
    for (var i in names) {
      for (var j in basename_level) {
        if (i === j) {
          flag = true;
        } else {
          flag = false;
        }
      }
    }

    if (flag && hashTable[data.basename_level] !== undefined) {
      return hashTable[data.basename_level].children.push(hashTable[data.name]);
    } else {
      return dataTree.push(hashTable[data.name]);
    }
  });

  children = [
    ...children,
    ...dataTree,
    { name: dataset.error.id },
    { name: dataset.error.message }
  ];
  const mainTree = {
    name: "picture data",
    toggled: false,
    children
  };
  return mainTree;
};

export default treeReducer;

        public static String getIDs(File file, String title) {
                String searchKey = "";
                String res = "";
                String titleValue = "";
                try (Scanner scanner = new Scanner(new File(file.getPath()))) {
                    scanner.useDelimiter("[,\n]");
                    while (scanner.hasNext()) {
                        searchKey = scanner.next();
                        if (!searchKey.contains(title)) {
                            continue;
                        }
                        titleValue = scanner.next();
                        res = titleValue;
                        break;
                    }
                }
                catch (FileNotFoundException e) {
                    LOGGER.log(Level.INFO, "name" + e);
                }
                return res;
            }

                boolean found = false;
                String id = "";
                String name = "";
                String age = "";
                try {
                    x = new Scanner(new File(filePath));
                    LOGGER.log(Level.INFO, "this is path" + filePath);
                    x.useDelimiter("[,\n]");
                    while(x.hasNext() && !found) {
                        id = x.next();
                        LOGGER.log(Level.INFO, "this is id..." + id);
                        name = "name";
                        age = "14";
                        LOGGER.log(Level.INFO, "searchTerm..." + searchTerm);
                        if(id.contains(searchTerm)) {
                            found = true;
                            LOGGER.log(Level.INFO, "found" + found);
                        }

                        if(found) {
                            JOptionPane.showMessageDialog(null, "ID: " +id + " Name: " + name + " Age: " + age);

                            LOGGER.log(Level.INFO, "Record not found");
                        }
                    }
