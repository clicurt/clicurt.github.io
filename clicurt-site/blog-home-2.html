$('button').on('click', function () { var timeOfCall = $('#timeOfCall').val(), timeOfResponse = $('#timeOfResponse').val(), hours = timeOfResponse.split(':')[0] - timeOfCall.split(':')[0], minutes = timeOfResponse.split(':')[1] - timeOfCall.split(':')[1]; minutes = minutes.toString().length<2?'0'+minutes:minutes; if(minutes<0){ hours--; minutes = 60 + minutes; } hours = hours.toString().length<2?'0'+hours:hours; $('#delay').val(hours + ':' + minutes); });
